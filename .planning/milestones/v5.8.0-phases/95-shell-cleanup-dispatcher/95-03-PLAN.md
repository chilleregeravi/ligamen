    wait_for_worker() {
      local max_attempts="${1:-20}"
      local interval_ms="${2:-250}"
      # DSP-09: pre-compute sleep interval once outside the loop.
      # awk is POSIX-available; single fork total, not max_attempts forks.
      local interval_s
      interval_s=$(awk -v ms="$interval_ms" 'BEGIN { printf "%.3f", ms/1000 }')
      local i=0
      while [[ $i -lt $max_attempts ]]; do
        if worker_running; then
          return 0
        fi
        sleep "$interval_s"
        i=$((i + 1))
      done
      echo "worker-client: timed out waiting for worker after $((max_attempts * interval_ms))ms" >&2
      return 1
    }
