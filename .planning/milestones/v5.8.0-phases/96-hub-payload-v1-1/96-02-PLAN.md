export function buildFindingsBlock(findings, opts = {}) -> {

  payload: {
    version: "1.0" | "1.1",      // === findingsBlock.schemaVersion
    metadata: {...},
    findings: {

      connections, schemas, actors,
    },
  },
  warnings: string[],
};

    // A representative DependencyRow shape — matches Phase 93-02

    const SAMPLE_DEP_ROW = {
      id: 1,
      service_id: 42,
      scan_version_id: 7,
      ecosystem: "npm",
      package_name: "react",
      version_spec: "^18.2.0",
      resolved_version: "18.2.0",
      manifest_file: "package.json",
      dep_kind: "direct",

    test("HUB-05 regression guard: buildFindingsBlock with no opts returns schemaVersion='1.0' and omits per-service dependencies", () => {
      const block = buildFindingsBlock({
        services: [{ name: "svc-a", language: "ts" }],
        connections: [{ source: "svc-a", target: "db", protocol: "tcp" }],
      });
      // Default path MUST be v1.0 — proves backward compat for every caller that
      // never opts into the feature flag.

      // Shape regression — existing keys are still present and unchanged.

      assert.equal(block.services[0].root_path, ".");
      assert.equal(block.services[0].type, "service");
      assert.ok(Array.isArray(block.connections));
      assert.ok(Array.isArray(block.schemas));
      assert.ok(Array.isArray(block.actors));
    });

    test("HUB-04 matrix #1: flag OFF + empty deps → schemaVersion='1.0', no per-service dependencies", () => {

            { name: "svc-b", language: "py", dependencies: [] },
          ],
          connections: [],
        },
        { libraryDepsEnabled: true },
      );
      assert.equal(block.schemaVersion, "1.0", "flag-on with all-empty deps must fall back to v1.0");
      assert.equal(block.services[0].dependencies, undefined, "v1.0 fallback must NOT emit per-service dependencies keys");
      assert.equal(block.services[1].dependencies, undefined);
    });

    test("HUB-04 matrix #3: flag OFF + populated deps → schemaVersion='1.0', deps suppressed (flag is authoritative)", () => {
      const block = buildFindingsBlock(
        {
          services: [{ name: "svc-a", language: "ts", dependencies: [SAMPLE_DEP_ROW] }],
          connections: [],
        },
        { libraryDepsEnabled: false },
      );
      assert.equal(block.schemaVersion, "1.0", "flag OFF dominates — dep data is ignored");
      assert.equal(block.services[0].dependencies, undefined, "v1.0 must NOT leak the input dependencies array");

    test("HUB-04 matrix #4: flag ON + populated deps → schemaVersion='1.1', per-service dependencies present", () => {
      const block = buildFindingsBlock(
        {
          services: [{ name: "svc-a", language: "ts", dependencies: [SAMPLE_DEP_ROW] }],
          connections: [],
        },
        { libraryDepsEnabled: true },
      );
      assert.equal(block.schemaVersion, "1.1");
      assert.ok(Array.isArray(block.services[0].dependencies), "v1.1 services must carry a dependencies array");
      assert.equal(block.services[0].dependencies.length, 1);
      assert.deepEqual(

        SAMPLE_DEP_ROW,
        "dep row shape must be preserved — payload.js does not reshape dep rows",
      );
    });

    test("HUB-04 matrix #5 (mixed services): flag ON + one service with deps + one without → v1.1 with dependencies on both (empty array on the one without)", () => {
      const block = buildFindingsBlock(

            { name: "svc-has-deps", language: "ts", dependencies: [SAMPLE_DEP_ROW] },

    });

      assert.ok(

      assert.equal(payload.findings.services[0].dependencies[0].package_name, "react");
