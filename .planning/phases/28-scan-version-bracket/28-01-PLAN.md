export const version = 5;
export function up(db) {

    CREATE TABLE IF NOT EXISTS scan_versions (
      id           INTEGER PRIMARY KEY AUTOINCREMENT,
      repo_id      INTEGER NOT NULL REFERENCES repos(id),
      started_at   TEXT    NOT NULL,
      completed_at TEXT
    );
