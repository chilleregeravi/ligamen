    ```
    /**
     * Write a row to enrichment_log.
     * No-op (returns null) when migration 015 not applied.
     * Throws if SQL CHECK constraint on target_kind fails.
     * @returns {number|null}
     */
    logEnrichment(scanVersionId, enricher, targetKind, targetId, field, fromValue, toValue, reason) {
      if (!this._stmtInsertEnrichmentLog) return null;
      const result = this._stmtInsertEnrichmentLog.run(
        scanVersionId, enricher, targetKind, targetId, field,
        fromValue ?? null, toValue ?? null, reason ?? null
      );
      return Number(result.lastInsertRowid);
    }

    /**
     * Read rows from enrichment_log for a scan_version_id.
     * @param {number} scanVersionId
     * @param {{ enricher?: string }} [opts]
     * @returns {Array}
     */
    getEnrichmentLog(scanVersionId, opts = {}) {
      if (!this._stmtSelectEnrichmentLog) return [];
      try {
        if (opts.enricher) {
          return this._stmtSelectEnrichmentLogByEnricher.all(scanVersionId, opts.enricher);
        }
        return this._stmtSelectEnrichmentLog.all(scanVersionId);
      } catch {
        return [];
      }
    }
    ```

    ```
    for (const conn of (findings.connections || [])) {
      if (!conn._reconciliation) continue;
      // Look up the persisted connection_id by source/target/path/method
      const sourceRow = db.prepare(
        'SELECT id FROM services WHERE name = ? AND repo_id = ?'
    ```
