```javascript
/**
 * Canonicalize a connection path by replacing every `{xxx}` template variable
 * with `{_}`. Used by persistFindings to collapse template-variant connections
 * (TRUST-03). Returns null/empty unchanged so we don't mint a `{_}` for
 * paths the agent didn't claim.
 *
 * Out of scope this phase: Express `:id` style, OpenAPI named groups, JAX-RS
 * constraint suffixes. See .planning/phases/109-*/109-CONTEXT.md D-06.
 *
 * @param {string|null|undefined} path
 * @returns {string|null}
 */
export function canonicalizePath(path) {
  if (path == null) return null;
  if (path === '') return '';
  return path.replace(/\{[^/}]+\}/g, '{_}');
}
```

(4) **Add a private helper** `_mergePathTemplates(existingCsv, newTemplate)` for the comma-join dedup:

```javascript
_mergePathTemplates(existingCsv, newTemplate) {
  if (!newTemplate) return existingCsv ?? null;
  if (!existingCsv) return newTemplate;
  const parts = existingCsv.split(',').map((s) => s.trim()).filter(Boolean);
  if (parts.includes(newTemplate)) return existingCsv;
  parts.push(newTemplate);
  return parts.join(',');
}
```

```javascript
/**
 * Validate that `evidence` appears as a literal substring in `source_file`.
 * Returns { ok: true } on success or skip-with-warn cases (null/empty evidence,
 * null source_file, missing source_file on disk). Returns { ok: false, reason }
 * only when the evidence is a non-empty string AND the source_file resolves to
 * a readable file AND the substring is not found.
 *
 * Per .planning/phases/109-*/109-CONTEXT.md:
 *   D-03: whole-file substring check (no line_start window — schema doesn't carry one).
 *   D-04: literal substring, no whitespace/regex normalization.
 *   D-05: lenient on null/missing/unreadable — return ok:true with optional warning.
 *
 * @param {{ evidence?: string|null, source_file?: string|null }} conn
 * @param {string|null} repoRootPath - From repos.path; used to resolve relative source_file.
 * @returns {{ ok: boolean, warn?: string }}
 */
_validateEvidence(conn, repoRootPath) {
  const evidence = (conn.evidence ?? '').trim();
  if (!evidence) return { ok: true }; // D-05 case 1: agent didn't claim evidence

  const srcRel = conn.source_file;
  if (srcRel == null || srcRel === '') return { ok: true }; // D-05 case 2: null source_file

  // Resolve relative paths against repo root. Absolute paths used as-is.
  let abs = srcRel;
  if (!path.isAbsolute(srcRel) && repoRootPath) {
    abs = path.join(repoRootPath, srcRel);
  }

  let content;
  try {
    content = fs.readFileSync(abs, 'utf8');
  } catch (e) {
    // D-05 case 3: file missing or unreadable — warn but don't reject
    return {
      ok: true,
      warn: `[persistFindings] cannot validate evidence: source_file '${srcRel}' does not exist or is unreadable (${e.code || e.message})`,
    };
  }

  if (content.indexOf(evidence) !== -1) {
    return { ok: true };
  }
  return {
    ok: false,
    reason: `evidence not found in '${srcRel}': ${evidence.slice(0, 80)}${evidence.length > 80 ? '…' : ''}`,
  };
}
```
