       The relative import from worker/scan/ to worker/lib/ is: '../lib/logger.js'
       Use: import { createLogger } from '../lib/logger.js';

    2. Add module-level logger (null by default — no-op when not set):
       /** @type {ReturnType<typeof createLogger> | null} */
       let _logger = null;

    3. Add a setter function (exported, for injection from worker/index.js or tests):
       export function setScanLogger(logger) {
         _logger = logger;
       }

    4. Add a scan-local log helper at the top of scanRepos():
       function slog(level, msg, extra = {}) {
         if (_logger) _logger.log(level, msg, extra);
       }

    5. Add log calls at these points in scanRepos():
       - After the for...of loop starts and ctx is determined:
         if (ctx.mode === 'skip') {

    For the production console.error in loadMigrations():
